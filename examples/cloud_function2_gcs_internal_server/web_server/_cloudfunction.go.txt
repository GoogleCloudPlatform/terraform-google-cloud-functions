package helloworld

import (
	"fmt"
	"io/ioutil"
	"log"
	"net/http"
)

func webserver() {
	client := &http.Client{}
	req, err := http.NewRequest("GET", "http://10.0.8.3:8080", nil)
	if err != nil {
		log.Fatal(err)
	}
	resp, err := client.Do(req)
	if err != nil {
		log.Fatal(err)
	}
	defer resp.Body.Close()
	bodyText, err := ioutil.ReadAll(resp.Body)
	if err != nil {
		log.Fatal(err)
	}
	fmt.Printf("%s\n", bodyText)
}
